#!/bin/bash

# Update and install
sudo apt update
sudo apt install -y apache2 php php-mysql mysql-server

# Start services
sudo systemctl start apache2
sudo systemctl start mysql
sudo systemctl enable apache2
sudo systemctl enable mysql

# Create DB
sudo mysql -u root << 'EOF'
CREATE DATABASE IF NOT EXISTS login_system;
USE login_system;
CREATE TABLE IF NOT EXISTS users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE USER 'phpuser'@'localhost' IDENTIFIED BY 'phppass';
GRANT ALL PRIVILEGES ON login_system.* TO 'phpuser'@'localhost';
FLUSH PRIVILEGES;
EXIT
EOF

echo "✅ EC2 Setup Complete!"
echo "Now copy files: scp -i key.pem -r /home/rakib/demo-php/* ubuntu@IP:/tmp/demo-php/"
echo "Then run: sudo cp -r /tmp/demo-php/* /var/www/html/ && sudo chown -R www-data:www-data /var/www/html"
